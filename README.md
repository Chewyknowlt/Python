### Scape Characters
\\"     -> (Double Quote), e.g; print("Hamza\"")       -> Hamza"
\\'     -> (Single Quote), e.g; print("Hamza\'")       -> Hamza'
\t      -> tab = 4 spaces, e.g; print("Hamza\tArain")  -> Hamza   Arain
\n      -> take newline,   e.g; print("Hamza\nArain")  -> Hamza
                                                          Arain
